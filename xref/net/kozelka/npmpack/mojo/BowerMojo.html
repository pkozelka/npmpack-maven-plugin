22      /**
23       * The executable to run as grunt. We expect that it resides under node_modules/.bin.
24       */
25      @Parameter(defaultValue = "bower,bower.cmd", required = true, property = "bower.executable")
26      String bowerExecutables;

28      /**
29       * The execution is skipped if this is true.

31      @Parameter(defaultValue = "false", required = true, property = "bower.skip")
32      boolean skip;

34      /**
35       * When true, manual (or maintenance) mode is assumed. In this mode, the plugin is allowed to download artifacts from non-Maven repositories, like NPM registry or from Bower.
36       * Otherwise it fails if the needed artifact is not available in maven repository.
37       */
38      @Parameter(defaultValue = "false", required = true, property = "npmpack.manual")
39      boolean manualMode;

42      /**
43       * Create temporary bower configuration file <code>.bowerrc</code> , making bower gather the dependencies into
44       * the ${project.build.directory}/${project.build.finalName}/res directory.
45       *
46       * This option is not effective if the file already exists.
47       * @todo consider if this logic is ok; wouldn't simple overwriting be better?
48       */
49      @Parameter(defaultValue = "true")
50      boolean bowerrcAutocreate;
51  
52      /**
53       * Directory where bower will create its libraries unless existing .bowerrc is defined.
54       */
55      @Parameter(defaultValue = "${project.build.directory}/${project.build.finalName}/lib")
56      File bowerOutputDirectory;
57  
58      @Override
59      public void execute() throws MojoExecutionException, MojoFailureException {
60          try {
61              final Commandline commandline = new Commandline();
62              commandline.setWorkingDirectory(basedir);
63              final File localBin = new File(node_modules, ".bin");
64              commandline.setExecutable(new File(localBin, selectAlternative(bowerExecutables)).getAbsolutePath());
65              commandline.addArguments(new String[]{"install"});
66              if (skip) {
67                  getLog().info("Bower execution is skipped: " + CommandLineUtils.toString(commandline.getShellCommandline()));
68                  return;
69              }
70  
71              final File bowerrcFile = new File(basedir, ".bowerrc");
72              final boolean temporary = !bowerrcFile.exists();
73              if (! temporary) {
74                  final JsonObject bowerrc = readBowerRc(bowerrcFile);
75                  bowerOutputDirectory = new File(bowerrc.get("directory").getAsString());
76                  getLog().info(".bowerrc/directory=" + bowerOutputDirectory);
77              }
78              // TODO: check that current content of bowerOutputDirectory matches bower.json (use checksum of anonymized canonicalized clone)
79              // TODO: add the trick to avoid the need for calling bower in regular builds!
80              if (manualMode) {
81                  // TODO: make sure that bower is in node_modules ? Like, by using "npm install bower" ?
82                  if (temporary) {
83                      FileUtils.fileWrite(bowerrcFile, "{\n    directory: \""+ jsonEscape(bowerOutputDirectory.getAbsolutePath()) +"\"\n}");
84                      bowerrcFile.deleteOnExit();
85                  }
86                  executeCommandline("bower", commandline);
87                  if (temporary) {
88                      bowerrcFile.delete();
89                  }
90                  //TODO: package the files in bowerOutputDirectory and put in local/remote repo
91              } else {
92                  throw new MojoExecutionException("Bower artifact not found, cannot create one in regular mode. Add '-DmanualMode' to Maven commandline to prepare one.");
93              }
94          } catch (InterruptedException e) {
95              throw new MojoExecutionException(e.getMessage(), e);
96          } catch (CommandLineException e) {
97              throw new MojoExecutionException(e.getMessage(), e);
98          } catch (IOException e) {
99              throw new MojoExecutionException(e.getMessage(), e);
100         }
101     }

103     private JsonObject readBowerRc(File bowerrcFile) throws IOException {
104         final String text = FileUtils.fileRead(bowerrcFile);
105         // always call this, to at least validate that it is correct json format
106         final JsonParser parser = new JsonParser();
107         return (JsonObject) parser.parse(text);
108     }
