14  /**
15   * Executes grunt in order to perform grunt-based build.
16   *
17   * @author Petr Kozelka
18   */
19  @Mojo(name="grunt", defaultPhase = LifecyclePhase.PACKAGE, requiresProject = true, threadSafe = true)
20  public class GruntMojo extends AbstractNpmpackMojo {
21  
22      /**
23       * The name to be used in prefix of this grunt command execution
24       */
25      @Parameter(defaultValue = "grunt", required = true)
26      String taskName;
27  
28  
29      /**
30       * The executable to run as grunt. We expect that it resides under node_modules/.bin.
31       */
32      @Parameter(defaultValue = "grunt,grunt.cmd", required = true, property = "grunt.executable")
33      String gruntExecutables;
34  
35      /**
36       * The specific grunt command to execute. This is passed as the (whitespace separated) argument list to grunt.
37       */
38      @Parameter(defaultValue = "build", required = true)
39      String gruntCommand;
40  
41      /**
42       * The execution is skipped it this is true.
43       */
44      @Parameter(defaultValue = "false", required = true, property = "grunt.skip")
45      boolean skip;
46  
47      @Override
48      public void execute() throws MojoExecutionException, MojoFailureException {
49          try {
50              final Commandline commandline = new Commandline();
51              commandline.setWorkingDirectory(basedir);
52              final File localBin = new File(node_modules, ".bin");
53              commandline.setExecutable(new File(localBin, selectAlternative(gruntExecutables)).getAbsolutePath());
54              commandline.addArguments(gruntCommand.split("\\s+"));
55              if (skip) {
56                  getLog().info("Grunt execution is skipped: " + CommandLineUtils.toString(commandline.getShellCommandline()));
57              } else {
58                  executeCommandline(taskName, commandline);
59              }
60          } catch (InterruptedException e) {
61              throw new MojoExecutionException(e.getMessage(), e);
62          } catch (CommandLineException e) {
63              throw new MojoExecutionException(e.getMessage(), e);
64          }
65      }
66  }
