98      final class MyStreamConsumer implements StreamConsumer {
99          private final String prefix;
100         private final boolean isErr;
101         private int lineCount = 0;
102 
103         MyStreamConsumer(String prefix, boolean isErr) {
104             this.prefix = prefix;
105             this.isErr = isErr;
106         }
107 
108         @Override
109         public void consumeLine(String line) {
110             final String msg = prefix + line;
111             if (isErr) {
112                 getLog().warn(msg);
113             } else {
114                 getLog().info(msg);
115             }
116             lineCount++;
117         }

124     protected String readPackageJson(File packageJson, boolean anonymize) throws IOException {
125         final String text = FileUtils.fileRead(packageJson);
126         // always call this, to at least validate that it is correct json format
127         final JsonParser parser = new JsonParser();
128         final JsonObject root = (JsonObject) parser.parse(text);
129         //JSONML.toJSONObject(text);
130         if (anonymize) {
131             getLog().info("Anonymizing name and version in " + packageJson);
132             root.addProperty("name", "ANONYMIZED");
133             root.addProperty("version", "ANONYMIZED");
134             JsonUtils.sortJsonDeep(root);
135             final Gson gson = new GsonBuilder()
136                     .setPrettyPrinting()
137                     .create();
138             return gson.toJson(root);
139         } else {
140             return text.replace("\r\n","\n").replace("\r", "\n");
141         }
142     }

144     /**
145      * Selects configured executable.
146      * If it consists of two comma-separated strings, the first is used on non-windows OS and the second on Windows.
147      * If there is only one string, it is always used as a whole.
148      * @param configuredAlternatives one or two comma-separated alternatives by OS; non-windows comes first, windows is second
149      * @return selected alternative
150      * @throws CommandLineException -
151      */
152     protected String selectAlternative(String configuredAlternatives) throws CommandLineException {
153         final String[] executables = configuredAlternatives.split(",");
154         final String npmExecutable;
155         if (executables.length == 1) {
156             npmExecutable = executables[0];
157             getLog().warn("Only one executable specified. You may have problems or other platforms: " + configuredAlternatives);
158         } else if (executables.length == 0) {
159             throw new CommandLineException("at least one executable variant must be specified");
160         } else {
161             npmExecutable = executables[Os.isFamily(Os.FAMILY_WINDOWS) ? 1 : 0];
162         }
163         return npmExecutable;
164     }

166     /**
167      * Executes given commandline. The output of execution if prefixed with taskName;
168      * square braces surround stdout lines, exclamation marks surround stderr lines.
169      * Little stats is printed at the end.
170      * @param taskName  logical task name to be used in prefix
171      * @param commandline -
172      * @throws CommandLineException
173      * @throws InterruptedException
174      */
175     protected void executeCommandline(String taskName, Commandline commandline) throws CommandLineException, InterruptedException {
176         getLog().info(String.format(" :::%s::: executing %s", taskName, CommandLineUtils.toString(commandline.getShellCommandline())));
177         final MyStreamConsumer stdout = new MyStreamConsumer("   [" + taskName + "] ", false);
178         final MyStreamConsumer stderr = new MyStreamConsumer("!" + taskName + "! ", true);
179         final long startTime = System.currentTimeMillis();
180         final int exitCode = CommandLineUtils.executeCommandLine(commandline, stdout, stderr);
181         getLog().info(String.format(" :::%s::: finished - exitCode: %d, duration: %d millis, output lines: %d on stdout, %d on stderr",
182                 taskName,
183                 exitCode,
184                 System.currentTimeMillis() - startTime,
185                 stdout.getLineCount(),
186                 stderr.getLineCount()
187         ));
188         if (exitCode != 0) {
189             final String javaVersion = System.getProperty("java.version");
190             getLog().warn("Using java " + javaVersion + " from " + System.getProperty("java.home"));
191             if (javaVersion.startsWith("1.7.0_25")) {
192                 getLog().warn("Java 1.7.0. update 25 is known to cause problems in executing npm. Upgrade your jdk!");
193                 getLog().warn("See http://stackoverflow.com/a/19318378/455449 for details");
194             }
195             throw new CommandLineException(commandline + " has failed with exitCode = " + exitCode);
196         }
197     }

199     protected void npm(String taskName, String... arguments) throws InterruptedException, CommandLineException {
200         final Commandline commandline = new Commandline();
201         commandline.setWorkingDirectory(basedir);
202         commandline.setExecutable(selectAlternative(npmExecutables));
203         commandline.addArguments(arguments);
