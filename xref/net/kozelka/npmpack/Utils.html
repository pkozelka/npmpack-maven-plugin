23      public static byte[] md5sum(InputStream fis) throws IOException, NoSuchAlgorithmException {
24          try {
25              final MessageDigest md5 = MessageDigest.getInstance("MD5");
26              final byte[] buffer = new byte[1024];
27              int numRead;
28              do {
29                  numRead = fis.read(buffer);
30                  if (numRead > 0) {
31                      md5.update(buffer, 0, numRead);
32                  }
33              } while (numRead != -1);
34              return md5.digest();
35          } finally {
36              fis.close();
37          }
38      }

41          final StringBuilder result = new StringBuilder();
42          for (byte b : bytes) {

44          }
45          return result.toString();
