19      /**
20       * Sorts properties of a json object, recursively
21       * @param jo
22       */
23      public static void sortJsonDeep(JsonElement jo) {
24          final LinkedList<JsonObject> unsorted = new LinkedList<JsonObject>();
25          if (jo.isJsonObject()) {
26              unsorted.add(jo.getAsJsonObject());
27          } else if (jo.isJsonArray()) {
28              for (JsonElement item : jo.getAsJsonArray()) {
29                  if (item.isJsonObject()) {
30                      unsorted.add(item.getAsJsonObject());
31                  }
32              }
33          }
34          while (!unsorted.isEmpty()) {
35              sortJsonObject(unsorted.removeFirst(), unsorted);
36          }
37      }

39      private static void sortJsonObject(JsonObject jo, Collection<JsonObject> unsorted) {
40          final List<Map.Entry<String, JsonElement>> entries = new ArrayList<Map.Entry<String, JsonElement>>(jo.entrySet());
41          Collections.sort(entries, new Comparator<Map.Entry<String, JsonElement>>() {
42              @Override
43              public int compare(Map.Entry<String, JsonElement> o1, Map.Entry<String, JsonElement> o2) {
44                  return o1.getKey().compareTo(o2.getKey());
45              }
46          });
47          // remove unsorted entries, and process those that are themselves objects
48          for (Map.Entry<String, JsonElement> entry : entries) {
49              final JsonElement value = jo.remove(entry.getKey());
50              if (value.isJsonObject()) {
51                  unsorted.add(value.getAsJsonObject());
52              } else if (value.isJsonArray()) {
53                  final JsonArray ja = value.getAsJsonArray();
54                  for (JsonElement jsonElement : ja) {
55                      if (jsonElement.isJsonObject()) {
56                          unsorted.add(jsonElement.getAsJsonObject());
57                      }
58                  }
59              }
60          }
61          // add back in sorted order
62          for (Map.Entry<String, JsonElement> entry : entries) {
63              jo.add(entry.getKey(), entry.getValue());
64          }
65      }
